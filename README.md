## Installation ##

    $ npm install -s endicia

## Usage

```javascript
var Endicia = require('endicia');
var client = new Endicia({
    REQUESTER_ID: '<PLACEHOLDER>',
    ACCOUNT_ID: '<PLACEHOLDER>',
    PASSPHRASE: '<PLACEHOLDER>'
});
```
